# ESET
************
این کد هارو در ترمینال بزنید

git clone https://github.com/SAMIYAR-BANG/ESET.git

cd ESET

chmod +x bot

./bot install

----------------------------------------------
مشخصات خودتون رو وارد کنید

./bot cli

----------------------------------------------

برای ربات هلپر

./bot api

killall screen && killall bash && killalll tmux && cd ESET && screen ./bot cli

cd ESET && screen ./bot api
